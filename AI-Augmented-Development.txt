The Array: By putting ['.json', '.ant'], the file picker window will now show both types of files as selectable.

Common JSON pitfalls to remember:
Trailing Commas: Unlike JavaScript objects, JSON cannot have a comma after the last item. (e.g., "params": {},

    /**
     * switchTab
     * @param {string} tabName - The name of the tab to activate (e.g., 'events', 'sprites')
     */
    function switchTab(tabName) {
        // 1. Hide all sections by removing the 'active' class
        var sections = document.querySelectorAll('.tab-section');
        for (var i = 0; i < sections.length; i++) {
            sections[i].classList.remove('active');
        }
        
        // 2. Deactivate all buttons
        var buttons = document.querySelectorAll('.tab-btn');
        for (var j = 0; j < buttons.length; j++) {
            buttons[j].classList.remove('active');
        }
        
        // 3. Show the selected section and activate its button
        var targetSection = document.getElementById('sec-' + tabName);
        var targetButton = document.getElementById('btn-tab-' + tabName);
        
        if (targetSection && targetButton) {
            targetSection.classList.add('active');
            targetButton.classList.add('active');
        }
        
        // Special Hook: If we switch to sprites, we need to refresh the list
        if (tabName === 'sprites') {
            populateSpriteList();
        }
    }

/**
 * updatePreview
 * Draws the selected sprite image and the current anchor point
 */
function updatePreview() {
	console.log("updatePreview called");
    if (!currentSelectedSprite) return;
    
	console.log("for "+currentSelectedSprite.name);
    var img = assetImages[currentSelectedSprite.name];
    if (!img) return;

	console.log("found the img ");
    var cw = previewCanvas.width;
    var ch = previewCanvas.height;
    pCtx.clearRect(0, 0, cw, ch);

    // 1. Calculate "Scale to Fit" (Letterbox)
    var scale = Math.min(cw / img.width, ch / img.height);
    var drawW = img.width * scale;
    var drawH = img.height * scale;
    var offsetX = (cw - drawW) / 2;
    var offsetY = (ch - drawH) / 2;

    // 2. Draw the image
    pCtx.drawImage(img, offsetX, offsetY, drawW, drawH);

    // 3. Draw the Anchor Point
    // We take the sprite's anchor (relative to image) and scale it to the preview
    var anchorX = offsetX + (currentSelectedSprite.x * scale);
    var anchorY = offsetY + (currentSelectedSprite.y * scale);

    // Draw Crosshair
    pCtx.strokeStyle = "red";
    pCtx.fillStyle = "red";
    pCtx.lineWidth = 1;

    // Horizontal line
    pCtx.beginPath();
    pCtx.moveTo(anchorX - 10, anchorY);
    pCtx.lineTo(anchorX + 10, anchorY);
    pCtx.stroke();

    // Vertical line
    pCtx.beginPath();
    pCtx.moveTo(anchorX, anchorY - 10);
    pCtx.lineTo(anchorX, anchorY + 10);
    pCtx.stroke();

    // Center dot
    pCtx.beginPath();
    pCtx.arc(anchorX, anchorY, 3, 0, Math.PI * 2);
    pCtx.fill();
}

function updatePreview() {
	console.log("updatePreview called");
    if (!currentSelectedSprite) return;
    
	console.log("for "+currentSelectedSprite.name);
    var img = assetImages[currentSelectedSprite.name];
    if (!img) return;

    var cw = previewCanvas.width;
    var ch = previewCanvas.height;
    pCtx.clearRect(0, 0, cw, ch);

    // 1. Calculate "Scale to Fit" (Letterbox)
    var scale = Math.min(cw / img.width, ch / img.height);
    var drawW = img.width * scale;
    var drawH = img.height * scale;
    var offsetX = (cw - drawW) / 2;
    var offsetY = (ch - drawH) / 2;

    // 2. Draw the image
    pCtx.drawImage(img, offsetX, offsetY, drawW, drawH);

    // 3. Draw the Anchor Point
    // We take the sprite's anchor (relative to image) and scale it to the preview
    var anchorX = offsetX + (currentSelectedSprite.x * scale);
    var anchorY = offsetY + (currentSelectedSprite.y * scale);

    // Draw Crosshair
    pCtx.strokeStyle = "red";
    pCtx.fillStyle = "red";
    pCtx.lineWidth = 1;

    // Horizontal line
    pCtx.beginPath();
    pCtx.moveTo(anchorX - 10, anchorY);
    pCtx.lineTo(anchorX + 10, anchorY);
    pCtx.stroke();

    // Vertical line
    pCtx.beginPath();
    pCtx.moveTo(anchorX, anchorY - 10);
    pCtx.lineTo(anchorX, anchorY + 10);
    pCtx.stroke();

    // Center dot
    pCtx.beginPath();
    pCtx.arc(anchorX, anchorY, 3, 0, Math.PI * 2);

    function switchTab(tabName) {
        // 1. Hide all sections by removing the 'active' class
        var sections = document.querySelectorAll('.tab-section');
        for (var i = 0; i < sections.length; i++) {
            sections[i].classList.remove('active');
        }
        
        // 2. Deactivate all buttons
        var buttons = document.querySelectorAll('.tab-btn');
        for (var j = 0; j < buttons.length; j++) {
            buttons[j].classList.remove('active');
        }
        
        // 3. Show the selected section and activate its button
        var targetSection = document.getElementById('sec-' + tabName);
        var targetButton = document.getElementById('btn-tab-' + tabName);
        
        if (targetSection && targetButton) {
            targetSection.classList.add('active');
            targetButton.classList.add('active');
        }
        
        // Special Hook: If we switch to sprites, we need to refresh the list
        if (tabName === 'sprites') {
            populateSpriteList();
        }
		if (tabName === 'stage') {
			populateStageList();
		}

    }

    var currentSelectedSprite = null;

/**
 * updatePreview
 * Draws the selected sprite image and the current anchor point
 */
function updatePreview() {
	console.log("updatePreview called");
    if (!currentSelectedSprite) return;
    
	console.log("for "+currentSelectedSprite.name);
    var img = assetImages[currentSelectedSprite.name];
    if (!img) return;

	console.log("found the img ");
    var cw = previewCanvas.width;
    var ch = previewCanvas.height;
	// create grid of dark and light grey transparancy
	pCtx.fillStyle = "#cccccc"; // Mid grey
	pCtx.fillRect(0, 0, cw, ch);
	pCtx.fillStyle = "#e5e5e5"; // Light grey	
	
    console.log("found the img ");
    var cw = previewCanvas.width;
    var ch = previewCanvas.height;
    var gridSize = 20;

    // 1. Fill the base color (dark grey)
    pCtx.fillStyle = "#888";
    pCtx.fillRect(0, 0, cw, ch);

    // 2. Draw the checkerboard squares (light grey)
    pCtx.fillStyle = "#aaa";
    for (var x = 0; x < cw; x += gridSize) {
        for (var y = 0; y < ch; y += gridSize) {
            // Use the index of the square, not the pixel coordinate
            // (x / 20) gives us 0, 1, 2, 3...
            var gridX = x / gridSize;
            var gridY = y / gridSize;

            // If the sum of the coordinates is even, draw a square
            // This creates the alternating "checker" pattern
            if ((gridX + gridY) % 2 === 0) {
                pCtx.fillRect(x, y, gridSize, gridSize);
            }
        }
    }

    // 1. Calculate "Scale to Fit" (Letterbox)
    var scale = Math.min(cw / img.width, ch / img.height);
    var drawW = img.width * scale;
    var drawH = img.height * scale;
    var offsetX = (cw - drawW) / 2;
    var offsetY = (ch - drawH) / 2;

    // 2. Draw the image
    pCtx.drawImage(img, offsetX, offsetY, drawW, drawH);

    // 3. Draw the Anchor Point
    // We take the sprite's anchor (relative to image) and scale it to the preview
    var anchorX = offsetX + (currentSelectedSprite.x * scale);
    var anchorY = offsetY + (currentSelectedSprite.y * scale);

    // Draw Crosshair
    pCtx.strokeStyle = "red";
    pCtx.fillStyle = "red";
    pCtx.lineWidth = 1;

    // Horizontal line
    pCtx.beginPath();
    pCtx.moveTo(anchorX - 10, anchorY);
    pCtx.lineTo(anchorX + 10, anchorY);
    pCtx.stroke();

    // Vertical line
    pCtx.beginPath();
    pCtx.moveTo(anchorX, anchorY - 10);
    pCtx.lineTo(anchorX, anchorY + 10);
    pCtx.stroke();

    // Center dot
    pCtx.beginPath();
    pCtx.arc(anchorX, anchorY, 3, 0, Math.PI * 2);
    pCtx.fill();
}

        function loadImage(file) {
            return new Promise(function(resolve) {
                var reader = new FileReader();
                reader.onload = function(e) {
                    var img = new Image();
                    img.onload = function() { resolve(img); };
                    img.src = e.target.result;
                };
                reader.readAsDataURL(file);
            });
        }

	var stage = {

	var instanceOverrides = {}; // { "mannetje_0": false, "vrouwtje_1": true }, allows for temporary hiding
		
	
	function stageToggleVisibility(instanceId) 
	{
		var inst = stage.instances[instanceId];
		if (!inst) return;

		// 1. Update the Override list (The immediate state)
		instanceOverrides[instanceId] = !inst.visible;

		// 2. Update the Timeline (The persistent record)
		var frameTime = (parseInt(document.getElementById('frameNum').value) || 0) / (parseInt(document.getElementById('timelineFps').value) || 30);
		
		var newEvent = {
			time: frameTime,
			instance: instanceId,
			type: instanceOverrides[instanceId] ? "enter" : "leave",
			params: instanceOverrides[instanceId] ? { sprite: inst.sprite, x: inst.x, y: inst.y, z: inst.z } : {}
		};

		timelineEvents.push(newEvent);
		sortTimeline();
		updateStage();
	}
	
	function rebuildStageState() {
		// 1. Reset instances to clear the slate for this frame
		stage.instances = {}; 

		var frameVal = parseInt(document.getElementById('frameNum').value) || 0;
		var fpsVal = parseInt(document.getElementById('timelineFps').value) || 30;
		var currentTime = frameVal / fpsVal;

		// 2. Replay all events in chronological order up to now
		timelineEvents.forEach(evt => {
			if (evt.time <= currentTime) {
				applyEventToStage(evt);
			}
		});
		
		  // 3. The User Override Authority (Toggles visibility)
		for (var id in instanceOverrides) 
		{
			if (stage.instances[id]) {
				stage.instances[id].visible = instanceOverrides[id];
			}
		}
	}

	/**
	 * initializeStage
	 * Creates the starting state for all sprites based on the project file
	 */
	function initializeStage() {
		// Start with a completely empty instance list.
		// No library sprites should be here by default.
		stage.instances = {}; 
		
		// We ONLY keep 'video' if you want it as a permanent background 
		// that doesn't need an "enter" event. 
		// Otherwise, even 'video' should be handled by an event.
		stage.instances["video"] = { 
			sprite: "video", 
			visible: false, 
			opacity: 1, 
			scale: 1, 
			x: 0, y: 0, z: 0 
		};
	}

	function createInstanceName(spriteName) {
    // Count how many instances of this specific sprite currently exist
    var count = 0;
    for (var id in stage.instances) {
        if (id.startsWith(spriteName + "_")) {
            count++;
        }
    }
    return spriteName + "_" + count;
}

	/* applyEventToStage
	   looping through all events up to the current TIME
	   creates the right state of the animation..
		we encounter a number of events that will be handled in THIS
		function that modifies the stage instance list.
	*/
		
	function applyEventToStage(evt) {
		var instId = evt.instance;
		if (!instId) return;

		if (evt.type === "enter") {
			var spriteName = evt.params.sprite; 
			var spriteDNA = sprites.find(s => s.name === spriteName);
			
			// Logic for naming: if "new", generate "mannetje_0"
			var finalId = instId;
			if (instId === "new") {
				var count = 0;
				for (var id in stage.instances) {
					if (id.startsWith(spriteName + "_")) count++;
				}
				finalId = spriteName + "_" + count;
